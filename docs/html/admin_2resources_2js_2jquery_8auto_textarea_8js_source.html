jquery.autoTextarea.js
1 /*
2 * autoTextarea-文本框根据输入内容自适应高度
3 * 参数 maxHeight:null,//文本框是否自动撑高，默认：null，不自动撑高；如果自动撑高必须输入数值，该值作为文本框自动撑高的最大高度
4 * 参数 minHeight:$(this).height()//默认最小高度，也就是文本框最初的高度，当内容高度小于这个高度的时候，文本以这个高度显示
5 */
6 (function($){
7  $.fn.autoTextarea = function(options) {
8  var defaults={
9  maxHeight:null,//文本框是否自动撑高，默认：null，不自动撑高；如果自动撑高必须输入数值，该值作为文本框自动撑高的最大高度
10  minHeight:$(this).height() //默认最小高度，也就是文本框最初的高度，当内容高度小于这个高度的时候，文本以这个高度显示
11  };
12  var opts = $.extend({},defaults,options);
13  return $(this).each(function() {
14  $(this).bind("paste cut keydown keyup focus blur",function(){
15  var height,style=this.style;
16  this.style.height = opts.minHeight + 'px';
17  if (this.scrollHeight > opts.minHeight) {
18  if (opts.maxHeight && this.scrollHeight > opts.maxHeight) {
19  height = opts.maxHeight;
20  style.overflowY = 'scroll';
21  } else {
22  height = this.scrollHeight;
23  style.overflowY = 'hidden';
24  }
25  style.height = height + 'px';
26  }
27  });
28  });
29  };
30 })(jQuery);
