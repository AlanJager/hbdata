431  function _compile_tag($template_tag)
432  {
433  /* Matched comment. */
434  if (substr($template_tag, 0, 1) == '*' && substr($template_tag, -1) == '*')
435  return '';
436 
437  /* Split tag into two three parts: command, command modifiers and the arguments. */
438  if(! preg_match('~^(?:(' . $this->_num_const_regexp . '|' . $this->_obj_call_regexp . '|' . $this->_var_regexp
439  . '|\/?' . $this->_reg_obj_regexp . '|\/?' . $this->_func_regexp . ')(' . $this->_mod_regexp . '*))
440  (?:\s+(.*))?$
441  ~xs', $template_tag, $match)) {
442  $this->_syntax_error("unrecognized tag: $template_tag", E_USER_ERROR, __FILE__, __LINE__);
443  }
444 
445  $tag_command = $match[1];
446  $tag_modifier = isset($match[2]) ? $match[2] : null;
447  $tag_args = isset($match[3]) ? $match[3] : null;
448 
449  if (preg_match('~^' . $this->_num_const_regexp . '|' . $this->_obj_call_regexp . '|' . $this->_var_regexp . '$~', $tag_command)) {
450  /* tag name is a variable or object */
451  $_return = $this->_parse_var_props($tag_command . $tag_modifier);
452  return "<?php echo $_return; ?>" . $this->_additional_newline;
453  }
454 
455  /* If the tag name is a registered object, we process it. */
456  if (preg_match('~^\/?' . $this->_reg_obj_regexp . '$~', $tag_command)) {
457  return $this->_compile_registered_object_tag($tag_command, $this->_parse_attrs($tag_args), $tag_modifier);
458  }
459 
460  switch ($tag_command) {
461  case 'include':
462  return $this->_compile_include_tag($tag_args);
463 
464  case 'include_php':
465  return $this->_compile_include_php_tag($tag_args);
466 
467  case 'if':
468  $this->_push_tag('if');
469  return $this->_compile_if_tag($tag_args);
470 
471  case 'else':
472  list($_open_tag) = end($this->_tag_stack);
473  if ($_open_tag != 'if' && $_open_tag != 'elseif')
474  $this->_syntax_error('unexpected {else}', E_USER_ERROR, __FILE__, __LINE__);
475  else
476  $this->_push_tag('else');
477  return '<?php else: ?>';
478 
479  case 'elseif':
480  list($_open_tag) = end($this->_tag_stack);
481  if ($_open_tag != 'if' && $_open_tag != 'elseif')
482  $this->_syntax_error('unexpected {elseif}', E_USER_ERROR, __FILE__, __LINE__);
483  if ($_open_tag == 'if')
484  $this->_push_tag('elseif');
485  return $this->_compile_if_tag($tag_args, true);
486 
487  case '/if':
488  $this->_pop_tag('if');
489  return '<?php endif; ?>';
490 
491  case 'capture':
492  return $this->_compile_capture_tag(true, $tag_args);
493 
494  case '/capture':
495  return $this->_compile_capture_tag(false);
496 
497  case 'ldelim':
498  return $this->left_delimiter;
499 
500  case 'rdelim':
501  return $this->right_delimiter;
502 
503  case 'section':
504  $this->_push_tag('section');
505  return $this->_compile_section_start($tag_args);
506 
507  case 'sectionelse':
508  $this->_push_tag('sectionelse');
509  return "<?php endfor; else: ?>";
510  break;
511 
512  case '/section':
513  $_open_tag = $this->_pop_tag('section');
514  if ($_open_tag == 'sectionelse')
515  return "<?php endif; ?>";
516  else
517  return "<?php endfor; endif; ?>";
518 
519  case 'foreach':
520  $this->_push_tag('foreach');
521  return $this->_compile_foreach_start($tag_args);
522  break;
523 
524  case 'foreachelse':
525  $this->_push_tag('foreachelse');
526  return "<?php endforeach; else: ?>";
527 
528  case '/foreach':
529  $_open_tag = $this->_pop_tag('foreach');
530  if ($_open_tag == 'foreachelse')
531  return "<?php endif; unset(\$_from); ?>";
532  else
533  return "<?php endforeach; endif; unset(\$_from); ?>";
534  break;
535 
536  case 'strip':
537  case '/strip':
538  if (substr($tag_command, 0, 1)=='/') {
539  $this->_pop_tag('strip');
540  if (--$this->_strip_depth==0) { /* outermost closing {/strip} */
541  $this->_additional_newline = "\n";
542  return '{' . $tag_command . '}';
543  }
544  } else {
545  $this->_push_tag('strip');
546  if ($this->_strip_depth++==0) { /* outermost opening {strip} */
547  $this->_additional_newline = "";
548  return '{' . $tag_command . '}';
549  }
550  }
551  return '';
552
