320  var getViewportHeight = function(){
321  var height = 0;
322  // first determine which children (slides) should be used in our height calculation
323  var children = $();
324  // if mode is not "vertical" and adaptiveHeight is false, include all children
325  if(slider.settings.mode != 'vertical' && !slider.settings.adaptiveHeight){
326  children = slider.children;
327  }else{
328  // if not carousel, return the single active child
329  if(!slider.carousel){
330  children = slider.children.eq(slider.active.index);
331  // if carousel, return a slice of children
332  }else{
333  // get the individual slide index
334  var currentIndex = slider.settings.moveSlides == 1 ? slider.active.index : slider.active.index * getMoveBy();
335  // add the current slide to the children
336  children = slider.children.eq(currentIndex);
337  // cycle through the remaining "showing" slides
338  for (i = 1; i <= slider.settings.maxSlides - 1; i++){
339  // if looped back to the start
340  if(currentIndex + i >= slider.children.length){
341  children = children.add(slider.children.eq(i - 1));
342  }else{
343  children = children.add(slider.children.eq(currentIndex + i));
344  }
345  }
346  }
347  }
348  // if "vertical" mode, calculate the sum of the heights of the children
349  if(slider.settings.mode == 'vertical'){
350  children.each(function(index) {
351  height += $(this).outerHeight();
352  });
353  // add user-supplied margins
354  if(slider.settings.slideMargin > 0){
355  height += slider.settings.slideMargin * (slider.settings.minSlides - 1);
356  }
357  // if not "vertical" mode, calculate the max height of the children
358  }else{
359  height = Math.max.apply(Math, children.map(function(){
360  return $(this).outerHeight(false);
361  }).get());
362  }

364  if(slider.viewport.css('box-sizing') == 'border-box'){
365  height += parseFloat(slider.viewport.css('padding-top')) + parseFloat(slider.viewport.css('padding-bottom')) +

1038  el.goToSlide = function(slideIndex, direction){
1039  // if plugin is currently in motion, ignore request
1040  if(slider.working || slider.active.index == slideIndex) return;
1041  // declare that plugin is in motion
1042  slider.working = true;
1043  // store the old index
1044  slider.oldIndex = slider.active.index;
1045  // if slideIndex is less than zero, set active index to last child (this happens during infinite loop)
1046  if(slideIndex < 0){
1047  slider.active.index = getPagerQty() - 1;
1048  // if slideIndex is greater than children length, set active index to 0 (this happens during infinite loop)
1049  }else if(slideIndex >= getPagerQty()){
1050  slider.active.index = 0;
1051  // set active index to requested slide
1052  }else{
1053  slider.active.index = slideIndex;
1054  }
1055  // onSlideBefore, onSlideNext, onSlidePrev callbacks
1056  slider.settings.onSlideBefore(slider.children.eq(slider.active.index), slider.oldIndex, slider.active.index);
1057  if(direction == 'next'){
1058  slider.settings.onSlideNext(slider.children.eq(slider.active.index), slider.oldIndex, slider.active.index);
1059  }else if(direction == 'prev'){
1060  slider.settings.onSlidePrev(slider.children.eq(slider.active.index), slider.oldIndex, slider.active.index);
1061  }
1062  // check if last slide
1063  slider.active.last = slider.active.index >= getPagerQty() - 1;
1064  // update the pager with active class
1065  if(slider.settings.pager) updatePagerActive(slider.active.index);
1066  // // check for direction control update
1067  if(slider.settings.controls) updateDirectionControls();
1068  // if slider is set to mode: "fade"
1069  if(slider.settings.mode == 'fade'){
1070  // if adaptiveHeight is true and next height is different from current height, animate to the new height
1071  if(slider.settings.adaptiveHeight && slider.viewport.height() != getViewportHeight()){
1072  slider.viewport.animate({height: getViewportHeight()}, slider.settings.adaptiveHeightSpeed);
1073  }

1075  slider.children.filter(':visible').fadeOut(slider.settings.speed).css({zIndex: 0});
1076  // fade in the newly requested slide
1077  slider.children.eq(slider.active.index).css('zIndex', slider.settings.slideZIndex+1).fadeIn(slider.settings.speed, function(){
1078  $(this).css('zIndex', slider.settings.slideZIndex);
1079  updateAfterSlideTransition();
1080  });
1081  // slider mode is not "fade"
1082  }else{
1083  // if adaptiveHeight is true and next height is different from current height, animate to the new height
1084  if(slider.settings.adaptiveHeight && slider.viewport.height() != getViewportHeight()){
1085  slider.viewport.animate({height: getViewportHeight()}, slider.settings.adaptiveHeightSpeed);
1086  }
1087  var moveBy = 0;
1088  var position = {left: 0, top: 0};
1089  // if carousel and not infinite loop
1090  if(!slider.settings.infiniteLoop && slider.carousel && slider.active.last){

1092  // get the last child position
1093  var lastChild = slider.children.eq(slider.children.length - 1);
1094  position = lastChild.position();
1095  // calculate the position of the last slide
1096  moveBy = slider.viewport.width() - lastChild.outerWidth();
1097  }else{
1098  // get last showing index position
1099  var lastShowingIndex = slider.children.length - slider.settings.minSlides;
1100  position = slider.children.eq(lastShowingIndex).position();
