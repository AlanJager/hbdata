10 KindEditor.plugin('lineheight', function(K) {
11  var self = this, name = 'lineheight', lang = self.lang(name + '.');
12  self.clickToolbar(name, function() {
13  var curVal = '', commonNode = self.cmd.commonNode({'*' : '.line-height'});
14  if (commonNode) {
15  curVal = commonNode.css('line-height');
16  }
17  var menu = self.createMenu({
18  name : name,
19  width : 150
20  });
21  K.each(lang.lineHeight, function(i, row) {
22  K.each(row, function(key, val) {
23  menu.addItem({
24  title : val,
25  checked : curVal === key,
26  click : function() {
27  self.cmd.toggle('<span style="line-height:' + key + ';"></span>', {
28  span : '.line-height=' + key
29  });
30  self.updateState();
31  self.addBookmark();
32  self.hideMenu();
33  }
34  });
35  });
36  });
37  });
38 });
