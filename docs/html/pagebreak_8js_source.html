10 KindEditor.plugin('pagebreak', function(K) {
11  var self = this;
12  var name = 'pagebreak';
13  var pagebreakHtml = K.undef(self.pagebreakHtml, '<hr style="page-break-after: always;" class="ke-pagebreak" />');
14 
15  self.clickToolbar(name, function() {
16  var cmd = self.cmd, range = cmd.range;
17  self.focus();
18  var tail = self.newlineTag == 'br' || K.WEBKIT ? '' : '<span id="__kindeditor_tail_tag__"></span>';
19  self.insertHtml(pagebreakHtml + tail);
20  if (tail !== '') {
21  var p = K('#__kindeditor_tail_tag__', self.edit.doc);
22  range.selectNodeContents(p[0]);
23  p.removeAttr('id');
24  cmd.select();
25  }
26  });
27 });
